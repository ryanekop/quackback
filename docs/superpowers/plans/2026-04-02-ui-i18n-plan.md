> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add UI string translation to the widget and public portal using FormatJS, with locale auto-detection and structural RTL support.

**Architecture:** Two independent `IntlProvider` mount points (portal root layout, widget auth provider). English strings bundled, other locales lazy-loaded via dynamic import. Widget accepts locale via SDK postMessage or URL param. Portal detects locale from `Accept-Language` header.

**Tech Stack:** react-intl (FormatJS), @formatjs/cli, Vitest, TanStack Start (SSR loaders for locale detection)

**Spec:** `docs/superpowers/specs/2026-04-02-ui-i18n-design.md`

Create `apps/web/src/lib/shared/i18n.ts`:

```typescript
export const DEFAULT_LOCALE = 'en'

export const SUPPORTED_LOCALES = ['en', 'de', 'pt', 'fr', 'es', 'it', 'ar'] as const

export type SupportedLocale = (typeof SUPPORTED_LOCALES)[number]

const RTL_LOCALES = new Set<string>(['ar', 'he', 'fa', 'ur'])

const supportedSet = new Set<string>(SUPPORTED_LOCALES)

/**
 * Normalize a locale tag to a supported locale.
 * Strips region codes (fr-FR -> fr) and lowercases.
 * Returns null if no supported locale matches.
 */
export function normalizeLocale(locale: string): SupportedLocale | null {
  if (!locale) return null
  const lower = locale.toLowerCase()
  // Exact match
  if (supportedSet.has(lower)) return lower as SupportedLocale
  // Strip region (fr-FR -> fr)
  const base = lower.split('-')[0]
  if (base && supportedSet.has(base)) return base as SupportedLocale
  return null
}

/**
 * Resolve the best locale from an Accept-Language header string.
 * An explicit override (e.g. from SDK or localStorage) takes precedence.
 */
export function resolveLocale(
  acceptLanguage: string | null | undefined,
  explicitLocale?: string
): SupportedLocale {
  // Explicit override takes precedence if it's a supported locale
  if (explicitLocale) {
    const normalized = normalizeLocale(explicitLocale)
    if (normalized) return normalized
  }

  if (!acceptLanguage) return DEFAULT_LOCALE

  // Parse Accept-Language: "fr-FR,fr;q=0.9,en;q=0.8"
  const entries = acceptLanguage
    .split(',')
    .map((entry) => {
      const [tag, ...params] = entry.trim().split(';')
      const qParam = params.find((p) => p.trim().startsWith('q='))
      const q = qParam ? parseFloat(qParam.trim().slice(2)) : 1
      return { tag: tag.trim(), q: Number.isNaN(q) ? 0 : q }
    })
    .sort((a, b) => b.q - a.q)

  for (const { tag } of entries) {
    const normalized = normalizeLocale(tag)
    if (normalized) return normalized
  }

  return DEFAULT_LOCALE
}

export function isRtlLocale(locale: string): boolean {
  return RTL_LOCALES.has(locale)
}

/** Load messages for a locale. English is returned synchronously from the bundled import. */
export async function loadMessages(locale: SupportedLocale): Promise<Record<string, string>> {
  if (locale === 'en') {
    const mod = await import('../../locales/en.json')
    return mod.default
  }
  try {
    const mod = await import(`../../locales/${locale}.json`)
    return mod.default
  } catch {
    // Locale file not found, fall back to English
    const en = await import('../../locales/en.json')
    return en.default
  }
}
```

interface PortalIntlProviderProps {

  children: ReactNode

export function PortalIntlProvider({ locale, children }: PortalIntlProviderProps) {

}
